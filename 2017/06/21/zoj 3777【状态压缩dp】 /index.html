zoj 3777【状态压缩dp】

题目链接：http://acm.zju.edu.cn/onlinejudge/showProblem.do?problemCode=3777

题意：有n到题目，当第i到题目放在第j个位置时有p[i][j]点乐趣，问有多少种排列方法使n个题目的乐趣和大于m

题解：

一开始想用dfs，复杂度一算为 12！ 铁定超时，后来就想到用状态压缩dp，但是我一开始想到的是dp[i][j][k]
表示前i行，放置列状态为j，分数为k的组合数，后来一写才发现转移过程时间复杂度为 nn2^n*m 最大可达2e8，所以就没写了。

后来看别人的题解才知道，可以降一维变成dp[i][j]表示列状态为i，分数为k的组合数。因为知道了列状态中的1的数目我们就可以知道放置了几行了，所以可以将第
一维舍去。

最终转移方程为

dp[i|(1<<k)][j+cnt] += dp[i][j] (j+cnt<m)

dp[i|(1<<k)][m] += dp[i][j] (j+cnt>=m)

(cnt为状态i中放置列数目)

#include<iostream>
#include<stdio.h>
#include<algorithm>
#include<cmath>
#include<stdlib.h>
#include<string.h>
#include<queue>
#include<set>
#include<map>
#include<stack>
#include<time.h>
#include<sstream>

using namespace std;
#define MAX_N 100005
#define inf 0x3f3f3f3f
#define LL long  long
#define ull unsigned long long
const LL INF = 1e18;
//const int mod = 1e9+7;
//const double pi = acos(-1);
typedef pair<int, int>P;

int dp[1<<13][505];
int gcd(int x, int y)
{
    return y==0?x:gcd(y, x%y);
}
int main()
{
    int T;
    cin >> T;
    while(T--)  {
        int n, m;
        cin >> n >> m;
        int p[15][15];
        for(int i=0; i<n; i++)
            for(int j=0; j<n; j++)
                scanf("%d", &p[i][j]);
        memset(dp, 0, sizeof(dp));
        dp[0][0] = 1;
        for(int i=0; i<(1<<n); i++) {
            int cnt = 0;
            for(int j=0; j<n; j++)
                if(i&(1<<j))
                    cnt++;
            for(int j=0; j<n; j++) {
                if(i&(1<<j))
                    continue;
                for(int k=0; k<=m; k++) {
                    if(k+p[cnt][j] < m)
                        dp[i|(1<<j)][k+p[cnt][j]] += dp[i][k];
                    else
                        dp[i|(1<<j)][m] += dp[i][k];
                }
            }
        }
        if(!dp[(1<<n)-1][m])
            puts("No solution");
        else {
            int tmp = 1;
            int ans = dp[(1<<n)-1][m];
            for(int i=2; i<=n; i++)
                tmp *= i;
            int g = gcd(tmp, ans);
            printf("%d/%d\n", tmp/g, ans/g);
        }
    }
}
